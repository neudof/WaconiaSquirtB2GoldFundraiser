  <script>

    function donateWithVenmo() {

      const amount = document.getElementById("donationAmount").value;

      if (amount >= 5) {

        window.open(`https://venmo.com/WaconiaWildcats_B2Gold?txn=pay&amount=${amount}&note=Donation+to+B2+Gold+Hockey`, "_blank");

      } else {

        alert("Please enter a donation amount of $5 or more.");

      }

    }


    document.getElementById("orderForm").addEventListener("submit", function(e) {

      e.preventDefault();

      const formData = new FormData(e.target);

      const data = Object.fromEntries(formData.entries());


      const payLater = document.getElementById("payLater").checked;

      data.paidWithVenmo = !payLater;

      data.amountPaid = payLater ? "0.00" : data.amount;


      fetch("YOUR_WEB_APP_URL", {

        method: "POST",

        body: JSON.stringify(data),

        headers: { "Content-Type": "application/json" }

      })

      .then(res => res.text())

      .then(msg => alert("Order submitted successfully!"))

      .catch(err => alert("Error submitting order."));

    });
